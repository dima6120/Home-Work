Expr ::= Let | Fun | expr
Let ::= 'let' Id '=' Expr 'in' Expr
Fun ::= 'fun' Id '->' Expr
expr ::= term (+|- term)
term ::= factor (*|/ factor)*
factor ::= funcall
funcall ::= primary | primary (primary)*
primary ::= Numb | Id | Brackets
Brackets ::= '(' Expr ')'
Id ::= Letter(Letter|Digit)*
Number ::= Digit(Digit)*
